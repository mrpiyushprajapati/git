class Solution {
public:
    int searchInsert(vector<int>& nums, int target) {
        int a;
        for(int i=0;i<nums.size();i++){
            if(nums[i]==target)
                a=i;
            else if(target>nums[i]&&target<nums[i+1])
                a=i+1;
        }
        return a;
    }
};
